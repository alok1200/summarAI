    27→    27→ * Lightweight video preview card shown inside the panel as soon as the user
    28→    28→ * pastes a valid URL. Fetches title + author + thumbnail from our own
    29→    29→ * /api/youtube-meta endpoint (which proxies YouTube's public oEmbed API).
    30→    30→ *
    31→    31→ * This gives the user visual confirmation that we picked up the right video
    32→    32→ * BEFORE they click Submit — avoiding wasted time on the wrong link.
    33→    33→ */
    34→    34→function VideoPreview({ videoId }: { videoId: string }) {
    35→    35→  // Start in the "loading" state. Because the parent renders this component
    36→    36→  // with `key={videoId}`, a new videoId causes React to remount this
    37→    37→  // component fresh — so the initial state is always "loading" for each new
    38→    38→  // video, without needing a synchronous setState inside useEffect.
    39→    39→  const [state, setState] = useState<
    40→    40→    | { kind: "loading" }
    41→    41→    | { kind: "ok"; title: string; author: string; thumbnailUrl: string }
    42→    42→    | { kind: "error"; message: string }
    43→    43→  >({ kind: "loading" });
    44→    44→
    45→    45→  useEffect(() => {
    46→    46→    let cancelled = false;
    47→    47→
    48→    48→    async function fetchMeta() {
    49→    49→      try {
    50→    50→        const res = await fetch(`/api/youtube-meta?videoId=${videoId}`);
    51→    51→        const data = await res.json();
    52→    52→        if (cancelled) return;
    53→    53→        if (!res.ok || !data.ok) {
    54→    54→          setState({
    55→    55→            kind: "error",
    56→    56→            message:
    57→    57→              data?.error ?? "Couldn't fetch metadata for this video.",
    58→    58→          });
    59→    59→          return;
    60→    60→        }
    61→    61→        setState({
    62→    62→          kind: "ok",
    63→    63→          title: data.title,
    64→    64→          author: data.author,
    65→    65→          thumbnailUrl: data.thumbnailUrl,
    66→    66→        });
    67→    67→      } catch (e) {
    68→    68→        if (cancelled) return;
    69→    69→        setState({
    70→    70→          kind: "error",
    71→    71→          message: "Network error while fetching video metadata.",
    72→    72→        });
    73→    73→      }
    74→    74→    }
    75→    75→
    76→    76→    fetchMeta();
    77→    77→    return () => {
    78→    78→      cancelled = true;
    79→    79→    };
    80→    80→  }, [videoId]);
    81→    81→
    82→    82→  if (state.kind === "loading") {
    83→    83→    return (
    84→    84→      <div className="flex items-center gap-2 rounded-lg border border-zinc-200 dark:border-zinc-700 bg-zinc-50 dark:bg-zinc-900 px-3 py-2 text-xs text-zinc-500">
    85→    85→        <Loader2 className="h-3.5 w-3.5 animate-spin" />
    86→    86→        Fetching video info…
    87→    87→      </div>
    88→    88→    );
    89→    89→  }
    90→    90→
    91→    91→  if (state.kind === "error") {
    92→    92→    return (
    93→    93→      <div className="flex items-center gap-2 rounded-lg border border-amber-200 dark:border-amber-800 bg-amber-50 dark:bg-amber-950/30 px-3 py-2 text-[11px] text-amber-700 dark:text-amber-300">
    94→    94→        <AlertCircle className="h-3.5 w-3.5 flex-shrink-0" />
    95→    95→        {state.message} You can still proceed — we&apos;ll fetch the transcript
    96→    96→        when you click Submit.
    97→    97→      </div>
    98→    98→    );
    99→    99→  }
   100→   100→
   101→   101→  return (
   102→   102→    <a
   103→   103→      href={`https://www.youtube.com/watch?v=${videoId}`}
   104→   104→      target="_blank"
   105→   105→      rel="noopener noreferrer"
   106→   106→      className="group flex gap-3 overflow-hidden rounded-lg border border-zinc-200 dark:border-zinc-700 bg-white dark:bg-zinc-900 p-2 hover:border-zinc-300 dark:hover:border-zinc-600 transition-colors"
   107→   107→    >
   108→   108→      <div className="relative h-14 w-24 flex-shrink-0 overflow-hidden rounded bg-zinc-100 dark:bg-zinc-800">
   109→   109→        <img
   110→   110→          src={state.thumbnailUrl}
   111→   111→          alt={state.title}
   112→   112→          className="h-full w-full object-cover"
   113→   113→          onError={(e) => {
   114→   114→            (e.currentTarget as HTMLImageElement).style.display = "none";
   115→   115→          }}
   116→   116→        />
   117→   117→        <div className="absolute inset-0 flex items-center justify-center bg-black/30 group-hover:bg-black/40 transition-colors">
   118→   118→          <Youtube className="h-5 w-5 text-white" />
   119→   119→        </div>
   120→   120→      </div>
   121→   121→      <div className="flex min-w-0 flex-1 flex-col justify-center">
   122→   122→        <p className="line-clamp-2 text-xs font-semibold text-zinc-800 dark:text-zinc-100 leading-snug">
   123→   123→          {state.title}
   124→   124→        </p>
   125→   125→        <p className="mt-0.5 truncate text-[11px] text-zinc-500 dark:text-zinc-400">
   126→   126→          {state.author}
   127→   127→        </p>
   128→   128→      </div>
   129→   129→    </a>
   130→   130→  );
   131→   131→}

   175→   175→function extractVideoId(url: string): string | null {
   176→   176→  const patterns: RegExp[] = [
   177→   177→    /(?:youtube\.com\/watch\?v=)([A-Za-z0-9_-]{11})/,
   178→   178→    /(?:youtu\.be\/)([A-Za-z0-9_-]{11})/,
   179→   179→    /(?:youtube\.com\/embed\/)([A-Za-z0-9_-]{11})/,
   180→   180→    /(?:youtube\.com\/shorts\/)([A-Za-z0-9_-]{11})/,
   181→   181→    /(?:youtube\.com\/live\/)([A-Za-z0-9_-]{11})/,
   182→   182→  ];
   183→   183→  for (const p of patterns) {
   184→   184→    const m = url.match(p);
   185→   185→    if (m) return m[1];
   186→   186→  }
   187→   187→  if (/^[A-Za-z0-9_-]{11}$/.test(url.trim())) return url.trim();
   188→   188→  return null;
   189→   189→}
