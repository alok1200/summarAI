    67‚Üíexport function parseTimeString(s: string): number | undefined {
    68‚Üí  const trimmed = s.trim();
    69‚Üí  if (!trimmed) return undefined;
    70‚Üí  // Support optional "m" / "s" / "h" unit suffixes for explicit input, e.g.:
    71‚Üí  //   "5m"     ‚Üí 5 minutes        "90s"   ‚Üí 90 seconds
    72‚Üí  //   "1h"     ‚Üí 1 hour           "1h30m" ‚Üí 1h30m
    73‚Üí  //   "2h15m30s" ‚Üí 2h15m30s       "1h 30m" ‚Üí 1h30m (spaces ok)
    74‚Üí  // These always win over the bare-number rule below.
    75‚Üí  if (/[hms]/i.test(trimmed) && /^[\d\s.:hms]+$/i.test(trimmed)) {
    76‚Üí    const h = trimmed.match(/(\d+)\s*h/i);
    77‚Üí    const m = trimmed.match(/(\d+)\s*m/i);
    78‚Üí    const sec = trimmed.match(/(\d+)\s*s/i);
    79‚Üí    if (h || m || sec) {
    80‚Üí      let total = 0;
    81‚Üí      if (h) total += parseInt(h[1], 10) * 3600;
    82‚Üí      if (m) total += parseInt(m[1], 10) * 60;
    83‚Üí      if (sec) total += parseInt(sec[1], 10);
    84‚Üí      return total;
    85‚Üí    }
    86‚Üí  }
    87‚Üí  const parts = trimmed.split(":").map((p) => parseInt(p, 10));
    88‚Üí  if (parts.some((n) => isNaN(n))) return undefined;
    89‚Üí  // BARE NUMBER ‚Üí MINUTES (so "5" means 5 min, not 5 sec).
    90‚Üí  // This matches how a human naturally thinks about video timestamps:
    91‚Üí  // "skip to 5" = the 5-minute mark, not 5 seconds in.
    92‚Üí  if (parts.length === 1) return parts[0] * 60;
    93‚Üí  if (parts.length === 2) return parts[0] * 60 + parts[1];
    94‚Üí  if (parts.length === 3) return parts[0] * 3600 + parts[1] * 60 + parts[2];
    95‚Üí  return undefined;
    96‚Üí}

    98‚Üíexport function formatTime(s: number): string {
    99‚Üí  const h = Math.floor(s / 3600);
   100‚Üí  const m = Math.floor((s % 3600) / 60);
   101‚Üí  const sec = Math.floor(s % 60);
   102‚Üí  const pad = (n: number) => n.toString().padStart(2, "0");
   103‚Üí  return h > 0 ? `${h}:${pad(m)}:${pad(sec)}` : `${m}:${pad(sec)}`;
   104‚Üí}

   278‚Üí    return cached;

   280‚Üí
   281‚Üí  // Strategy A: Official YouTube Data API v3 (most reliable ‚Äî no bot protection).
   282‚Üí  // Only used if YOUTUBE_API_KEY is set in the environment.
   283‚Üí  const apiKey = process.env.YOUTUBE_API_KEY;
   284‚Üí  if (apiKey && apiKey.trim() !== "") {
   285‚Üí    try {

   287‚Üí        `https://www.googleapis.com/youtube/v3/videos` +
   288‚Üí        `?part=snippet&id=${videoId}&key=${encodeURIComponent(apiKey)}`;
   289‚Üí      const res = await fetch(url, {
   290‚Üí        headers: { "Accept-Language": "en-US,en;q=0.9" },
   291‚Üí      });

   299‚Üí                high?: { url?: string };

   301‚Üí                default?: { url?: string };
   302‚Üí              };
   303‚Üí            };
   304‚Üí          }>;
   305‚Üí        };
   306‚Üí        const item = json.items?.[0];
   307‚Üí        if (item?.snippet?.title) {

   312‚Üí              item.snippet.thumbnails?.high?.url ||
   313‚Üí              item.snippet.thumbnails?.medium?.url ||
   314‚Üí              item.snippet.thumbnails?.default?.url ||
   315‚Üí              `https://img.youtube.com/vi/${videoId}/hqdefault.jpg`,

   319‚Üí          return entry;
   320‚Üí        }
   321‚Üí      }
   322‚Üí    } catch {

   326‚Üí
   327‚Üí  // Strategy B: Public oEmbed endpoint (no API key needed, but can fail on

   329‚Üí  try {

   333‚Üí    );
   334‚Üí    if (!res.ok) return null;
   335‚Üí    const json = (await res.json()) as {
   336‚Üí      title?: string;
   337‚Üí      author_name?: string;
   338‚Üí      thumbnail_url?: string;
   339‚Üí    };
   340‚Üí    if (!json.title) return null;
   341‚Üí    const entry: MetaCacheEntry = {
   342‚Üí      title: json.title,
   343‚Üí      author: json.author_name || "Unknown",
   344‚Üí      thumbnailUrl:
   345‚Üí        json.thumbnail_url ||
   346‚Üí        `https://img.youtube.com/vi/${videoId}/hqdefault.jpg`,
   347‚Üí      fetchedAt: Date.now(),
   348‚Üí    };
   349‚Üí    metaCache.set(videoId, entry);
   350‚Üí    return entry;
   351‚Üí  } catch {
   352‚Üí    return null;
   353‚Üí  }

   555‚Üífunction parseXmlTranscript(text: string): TranscriptSegment[] {
   556‚Üí  const segments: TranscriptSegment[] = [];
   557‚Üí  const srv3Regex = /<p\s+t="(\d+)"\s+d="(\d+)"[^>]*>([\s\S]*?)<\/p>/g;
   558‚Üí  let m: RegExpExecArray | null;
   559‚Üí  while ((m = srv3Regex.exec(text)) !== null) {
   560‚Üí    const start = parseInt(m[1], 10) / 1000;
   561‚Üí    const dur = parseInt(m[2], 10) / 1000;
   562‚Üí    const inner = m[3];
   563‚Üí    let segText = "";
   564‚Üí    const sRegex = /<s[^>]*>([^<]*)<\/s>/g;
   565‚Üí    let sMatch: RegExpExecArray | null;
   566‚Üí    while ((sMatch = sRegex.exec(inner)) !== null) {
   567‚Üí      segText += sMatch[1];
   568‚Üí    }
   569‚Üí    if (!segText) segText = inner.replace(/<[^>]+>/g, "");
   570‚Üí    segText = decodeEntities(segText).replace(/\n/g, " ").trim();
   571‚Üí    if (segText) segments.push({ start, dur, text: segText });
   572‚Üí  }
   573‚Üí  if (segments.length > 0) return segments;
   574‚Üí
   575‚Üí  const xmlRegex =
   576‚Üí    /<text\s+start="([\d.]+)"\s+dur="([\d.]+)"[^>]*>([\s\S]*?)<\/text>/g;
   577‚Üí  while ((m = xmlRegex.exec(text)) !== null) {
   578‚Üí    const start = parseFloat(m[1]);
   579‚Üí    const dur = parseFloat(m[2]);
   580‚Üí    const segText = decodeEntities(m[3])
   581‚Üí      .replace(/<[^>]+>/g, "")
   582‚Üí      .replace(/\n/g, " ")
   583‚Üí      .trim();
   584‚Üí    if (segText) segments.push({ start, dur, text: segText });
   585‚Üí  }
   586‚Üí  return segments;
   587‚Üí}

   749‚Üí/**
   750‚Üí * Parse a user-pasted transcript. Accepts:
   751‚Üí *   - "MM:SS text" or "HH:MM:SS text" (one per line)
   752‚Üí *   - "[MM:SS] text"
   753‚Üí *   - "1:23:45 text"
   754‚Üí *   - Plain text (no timestamps) ‚Äî each non-empty line becomes a segment
   755‚Üí *     with a sequential 5-second offset.
   756‚Üí */
   757‚Üíexport function parseUserTranscript(raw: string): {
   758‚Üí  segments: TranscriptSegment[];
   759‚Üí  hasTimestamps: boolean;
   760‚Üí} {
   761‚Üí  const lines = raw.split(/\r?\n/).map((l) => l.trim()).filter(Boolean);
   762‚Üí  if (lines.length === 0) return { segments: [], hasTimestamps: false };
   763‚Üí
   764‚Üí  const segments: TranscriptSegment[] = [];
   765‚Üí  let lastStart = 0;
   766‚Üí  let hasTimestamps = false;
   767‚Üí  const tsRegex = /^\[?(\d{1,2}:\d{2}(?::\d{2})?)\]?\s*(.*)$/;
   768‚Üí
   769‚Üí  for (const line of lines) {
   770‚Üí    const m = line.match(tsRegex);
   771‚Üí    if (m) {
   772‚Üí      const parts = m[1].split(":").map((p) => parseInt(p, 10));
   773‚Üí      let sec: number;
   774‚Üí      if (parts.length === 2) sec = parts[0] * 60 + parts[1];
   775‚Üí      else if (parts.length === 3) sec = parts[0] * 3600 + parts[1] * 60 + parts[2];
   776‚Üí      else sec = 0;
   777‚Üí      lastStart = sec;
   778‚Üí      hasTimestamps = true;
   779‚Üí      if (m[2]) {
   780‚Üí        segments.push({ start: sec, dur: 5, text: m[2].trim() });
   781‚Üí      }
   782‚Üí    } else {
   783‚Üí      segments.push({ start: lastStart, dur: 5, text: line });
   784‚Üí      lastStart += 5;
   785‚Üí    }
   786‚Üí  }
   787‚Üí
   788‚Üí  return { segments, hasTimestamps };
   789‚Üí}
