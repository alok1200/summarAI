   126Ôćĺexport function parseTimeString(s: string): number | undefined {
   127Ôćĺ  const trimmed = s.trim();
   128Ôćĺ  if (!trimmed) return undefined;
   129Ôćĺ  // Support optional "m" / "s" / "h" unit suffixes for explicit input, e.g.:
   130Ôćĺ  //   "5m"     Ôćĺ 5 minutes        "90s"   Ôćĺ 90 seconds
   131Ôćĺ  //   "1h"     Ôćĺ 1 hour           "1h30m" Ôćĺ 1h30m
   132Ôćĺ  //   "2h15m30s" Ôćĺ 2h15m30s       "1h 30m" Ôćĺ 1h30m (spaces ok)
   133Ôćĺ  // These always win over the bare-number rule below.
   134Ôćĺ  if (/[hms]/i.test(trimmed) && /^[\d\s.:hms]+$/i.test(trimmed)) {
   135Ôćĺ    const h = trimmed.match(/(\d+)\s*h/i);
   136Ôćĺ    const m = trimmed.match(/(\d+)\s*m/i);
   137Ôćĺ    const sec = trimmed.match(/(\d+)\s*s/i);
   138Ôćĺ    if (h || m || sec) {
   139Ôćĺ      let total = 0;
   140Ôćĺ      if (h) total += parseInt(h[1], 10) * 3600;
   141Ôćĺ      if (m) total += parseInt(m[1], 10) * 60;
   142Ôćĺ      if (sec) total += parseInt(sec[1], 10);
   143Ôćĺ      return total;
   144Ôćĺ    }
   145Ôćĺ  }
   146Ôćĺ  const parts = trimmed.split(":").map((p) => parseInt(p, 10));
   147Ôćĺ  if (parts.some((n) => isNaN(n))) return undefined;
   148Ôćĺ  // BARE NUMBER Ôćĺ MINUTES (so "5" means 5 min, not 5 sec).
   149Ôćĺ  // This matches how a human naturally thinks about video timestamps:
   150Ôćĺ  // "skip to 5" = the 5-minute mark, not 5 seconds in.
   151Ôćĺ  if (parts.length === 1) return parts[0] * 60;
   152Ôćĺ  if (parts.length === 2) return parts[0] * 60 + parts[1];
   153Ôćĺ  if (parts.length === 3) return parts[0] * 3600 + parts[1] * 60 + parts[2];
   154Ôćĺ  return undefined;
   155Ôćĺ}

   627Ôćĺfunction parseXmlTranscript(text: string): TranscriptSegment[] {
   628Ôćĺ  const segments: TranscriptSegment[] = [];
   629Ôćĺ  const srv3Regex = /<p\s+t="(\d+)"\s+d="(\d+)"[^>]*>([\s\S]*?)<\/p>/g;
   630Ôćĺ  let m: RegExpExecArray | null;
   631Ôćĺ  while ((m = srv3Regex.exec(text)) !== null) {
   632Ôćĺ    const start = parseInt(m[1], 10) / 1000;
   633Ôćĺ    const dur = parseInt(m[2], 10) / 1000;
   634Ôćĺ    const inner = m[3];
   635Ôćĺ    let segText = "";
   636Ôćĺ    const sRegex = /<s[^>]*>([^<]*)<\/s>/g;
   637Ôćĺ    let sMatch: RegExpExecArray | null;
   638Ôćĺ    while ((sMatch = sRegex.exec(inner)) !== null) {
   639Ôćĺ      segText += sMatch[1];
   640Ôćĺ    }
   641Ôćĺ    if (!segText) segText = inner.replace(/<[^>]+>/g, "");
   642Ôćĺ    segText = decodeEntities(segText).replace(/\n/g, " ").trim();
   643Ôćĺ    if (segText) segments.push({ start, dur, text: segText });
   644Ôćĺ  }
   645Ôćĺ  if (segments.length > 0) return segments;
   646Ôćĺ
   647Ôćĺ  const xmlRegex =
   648Ôćĺ    /<text\s+start="([\d.]+)"\s+dur="([\d.]+)"[^>]*>([\s\S]*?)<\/text>/g;
   649Ôćĺ  while ((m = xmlRegex.exec(text)) !== null) {
   650Ôćĺ    const start = parseFloat(m[1]);
   651Ôćĺ    const dur = parseFloat(m[2]);
   652Ôćĺ    const segText = decodeEntities(m[3])
   653Ôćĺ      .replace(/<[^>]+>/g, "")
   654Ôćĺ      .replace(/\n/g, " ")
   655Ôćĺ      .trim();
   656Ôćĺ    if (segText) segments.push({ start, dur, text: segText });
   657Ôćĺ  }
   658Ôćĺ  return segments;
   659Ôćĺ}

   701Ôćĺ/**
   702Ôćĺ * Try multiple strategies in order to fetch the transcript. Each strategy
   703Ôćĺ * has different bot-detection characteristics.
   704Ôćĺ *
   705Ôćĺ * Throws a `BotBlockedError` (with `code === "BOT_BLOCKED"`) if any strategy
   706Ôćĺ * reported a bot-block signature. Throws a generic Error otherwise.
   707Ôćĺ */
   708Ôćĺexport async function fetchTranscriptWithRetry(
   709Ôćĺ  videoId: string
   710Ôćĺ): Promise<TranscriptSegment[]> {
   711Ôćĺ  const cached = transcriptCache.get(videoId);
   712Ôćĺ  if (cached && Date.now() - cached.fetchedAt < CACHE_TTL_MS) {
   713Ôćĺ    return cached.segments;
   714Ôćĺ  }
   715Ôćĺ
   716Ôćĺ  const trackStrategies: {
   717Ôćĺ    name: string;
   718Ôćĺ    fn: (id: string) => Promise<CaptionTrack[] | null>;
   719Ôćĺ  }[] = [
   720Ôćĺ    {
   721Ôćĺ      name: "ANDROID player API",
   722Ôćĺ      fn: fetchCaptionTracksViaAndroidPlayer,
   723Ôćĺ    },
   724Ôćĺ    {
   725Ôćĺ      name: "Watch page scrape",
   726Ôćĺ      fn: fetchCaptionTracksViaWatchPage,
   727Ôćĺ    },
   728Ôćĺ  ];
   729Ôćĺ
   730Ôćĺ  const directStrategies: {
   731Ôćĺ    name: string;
   732Ôćĺ    fn: (id: string) => Promise<TranscriptSegment[] | null>;
   733Ôćĺ  }[] = [
   734Ôćĺ    {
   735Ôćĺ      name: "youtube-transcript library",
   736Ôćĺ      fn: fetchTranscriptViaYoutubeTranscriptLib,
   737Ôćĺ    },
   738Ôćĺ    {
   739Ôćĺ      name: "youtubei.js (Innertube)",
   740Ôćĺ      fn: fetchTranscriptViaInnertube,
   741Ôćĺ    },
   742Ôćĺ  ];
   743Ôćĺ
   744Ôćĺ  // Returns true if the error message looks like YouTube bot-protection /
   745Ôćĺ  // rate-limiting. Used both to flag the final error as BOT_BLOCKED and to
   746Ôćĺ  // decide whether to use a longer backoff before the next strategy.
   747Ôćĺ  function isBotBlockMessage(msg: string): boolean {
   748Ôćĺ    return /sign in|bot|consent|429|rate.?limit|captcha|too many requests|unusual traffic/i.test(
   749Ôćĺ      msg
   750Ôćĺ    );
   751Ôćĺ  }
   752Ôćĺ
   753Ôćĺ  let lastError: Error | null = null;
   754Ôćĺ  let botBlockedSeen = false;
   755Ôćĺ  let segments: TranscriptSegment[] | null = null;
   756Ôćĺ
   757Ôćĺ  for (const strategy of trackStrategies) {
   758Ôćĺ    try {
   759Ôćĺ      console.log(`[youtube] Trying ${strategy.name} for ${videoId}`);
   760Ôćĺ      const tracks = await strategy.fn(videoId);
   761Ôćĺ      if (!tracks || tracks.length === 0) {
   762Ôćĺ        console.log(`[youtube] ÔťŚ ${strategy.name} returned no tracks`);
   763Ôćĺ        continue;
   764Ôćĺ      }
   765Ôćĺ      console.log(
   766Ôćĺ        `[youtube] Ôťô ${strategy.name} returned ${tracks.length} tracks`
   767Ôćĺ      );
   768Ôćĺ      segments = await fetchAndParseCaptionTracks(tracks);
   769Ôćĺ      if (segments.length > 0) break;
   770Ôćĺ    } catch (e) {
   771Ôćĺ      const msg = (e as Error).message;
   772Ôćĺ      console.log(`[youtube] ÔťŚ ${strategy.name} failed: ${msg}`);
   773Ôćĺ      const blocked = isBotBlockMessage(msg);
   774Ôćĺ      if (blocked) {
   775Ôćĺ        botBlockedSeen = true;
   776Ôćĺ        lastError = e as Error;
   777Ôćĺ      } else if (!botBlockedSeen) {
   778Ôćĺ        lastError = e as Error;
   779Ôćĺ      }
   780Ôćĺ      // 429 = rate limit: give YouTube time to cool down before next attempt.
   781Ôćĺ      // Bot-block / "Sign in" usually clears with a 1.5s pause. Other errors
   782Ôćĺ      // (e.g. caption format unsupported) don't need any delay.
   783Ôćĺ      const backoff = blocked ? 1500 : 400;
   784Ôćĺ      await new Promise((r) => setTimeout(r, backoff));
   785Ôćĺ    }
   786Ôćĺ  }
   787Ôćĺ
   788Ôćĺ  if (!segments || segments.length === 0) {
   789Ôćĺ    for (const strategy of directStrategies) {
   790Ôćĺ      try {
   791Ôćĺ        console.log(`[youtube] Trying ${strategy.name} for ${videoId}`);

   848Ôćĺ/**
   849Ôćĺ * Parse a user-pasted transcript. Accepts:
   850Ôćĺ *   - "MM:SS text" or "HH:MM:SS text" (one per line)
   851Ôćĺ *   - "[MM:SS] text"
   852Ôćĺ *   - "1:23:45 text"
   853Ôćĺ *   - Plain text (no timestamps) ÔÇö each non-empty line becomes a segment
   854Ôćĺ *     with a sequential 5-second offset.
   855Ôćĺ */
   856Ôćĺexport function parseUserTranscript(raw: string): {
   857Ôćĺ  segments: TranscriptSegment[];
   858Ôćĺ  hasTimestamps: boolean;
   859Ôćĺ} {
   860Ôćĺ  const lines = raw.split(/\r?\n/).map((l) => l.trim()).filter(Boolean);
   861Ôćĺ  if (lines.length === 0) return { segments: [], hasTimestamps: false };
   862Ôćĺ
   863Ôćĺ  const segments: TranscriptSegment[] = [];
   864Ôćĺ  let lastStart = 0;
   865Ôćĺ  let hasTimestamps = false;
   866Ôćĺ  const tsRegex = /^\[?(\d{1,2}:\d{2}(?::\d{2})?)\]?\s*(.*)$/;
   867Ôćĺ
   868Ôćĺ  for (const line of lines) {
   869Ôćĺ    const m = line.match(tsRegex);
   870Ôćĺ    if (m) {

   872Ôćĺ      let sec: number;
   873Ôćĺ      if (parts.length === 2) sec = parts[0] * 60 + parts[1];
   874Ôćĺ      else if (parts.length === 3) sec = parts[0] * 3600 + parts[1] * 60 + parts[2];
   875Ôćĺ      else sec = 0;
   876Ôćĺ      lastStart = sec;
   877Ôćĺ      hasTimestamps = true;
   878Ôćĺ      if (m[2]) {
   879Ôćĺ        segments.push({ start: sec, dur: 5, text: m[2].trim() });
   880Ôćĺ      }
   881Ôćĺ    } else {
   882Ôćĺ      segments.push({ start: lastStart, dur: 5, text: line });
   883Ôćĺ      lastStart += 5;
   884Ôćĺ    }
   885Ôćĺ  }
   886Ôćĺ
   887Ôćĺ  return { segments, hasTimestamps };
   888Ôćĺ}
