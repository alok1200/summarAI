    55‚Üí * server. YouTube's IP-level rate limiter will then see the proxy's IP, not
    56‚Üí * ours ‚Äî which is the only client-side workaround for the "Sign in to
    57‚Üí * confirm you're not a bot" block that affects all 4 in-process strategies

    61‚Üí *   YOUTUBE_PROXY_URL="https://my-yt-proxy.example.workers.dev/?url="

    64‚Üí * Leave empty (the default) ‚Üí all requests go direct from this server's IP.
    65‚Üí *
    66‚Üí * NOTE: this only proxies OUR fetches. The youtube-transcript npm package and
    67‚Üí * youtubei.js make their own internal fetches that we can't intercept, so
    68‚Üí * those strategies still hit YouTube directly. The proxy primarily helps
    69‚Üí * strategies 1 (ANDROID) and 2 (watch-page) ‚Äî which are the most reliable
    70‚Üí * anyway.

   109‚Üí    return warmedCookies.cookieHeader;

   112‚Üí    const res = await proxiedFetch("https://www.youtube.com/", {

   115‚Üí        "Accept-Language": "en-US,en;q=0.9",

   117‚Üí          "text/html,application/xhtml+xml,application/xml;q=0.9,image/avif,image/webp,*/*;q=0.8",

   119‚Üí        "Sec-Fetch-Mode": "navigate",
   120‚Üí        "Sec-Fetch-Site": "none",
   121‚Üí        "Upgrade-Insecure-Requests": "1",

   123‚Üí      redirect: "follow",
   124‚Üí    });

   128‚Üí      const m = sc.match(/^([^=]+)=([^;]+)/);
   129‚Üí      if (m) pairs.push(`${m[1]}=${m[2]}`);
   130‚Üí    }
   131‚Üí    const header = pairs.length > 0 ? pairs.join("; ") : "";
   132‚Üí    warmedCookies = { cookieHeader: header, fetchedAt: Date.now() };

   134‚Üí      `[youtube] Warmed ${pairs.length} cookies: ${pairs
   135‚Üí        .map((p) => p.split("=")[0])
   136‚Üí        .join(", ")}`
   137‚Üí    );
   138‚Üí    return header;
   139‚Üí  } catch {
   140‚Üí    // Cookie warming is best-effort ‚Äî strategies still work without it,
   141‚Üí    // just with a higher block rate.
   142‚Üí    return "";
   143‚Üí  }

   176‚Üíexport function parseTimeString(s: string): number | undefined {
   177‚Üí  const trimmed = s.trim();
   178‚Üí  if (!trimmed) return undefined;
   179‚Üí  // Support optional "m" / "s" / "h" unit suffixes for explicit input, e.g.:
   180‚Üí  //   "5m"     ‚Üí 5 minutes        "90s"   ‚Üí 90 seconds
   181‚Üí  //   "1h"     ‚Üí 1 hour           "1h30m" ‚Üí 1h30m
   182‚Üí  //   "2h15m30s" ‚Üí 2h15m30s       "1h 30m" ‚Üí 1h30m (spaces ok)
   183‚Üí  // These always win over the bare-number rule below.
   184‚Üí  if (/[hms]/i.test(trimmed) && /^[\d\s.:hms]+$/i.test(trimmed)) {
   185‚Üí    const h = trimmed.match(/(\d+)\s*h/i);
   186‚Üí    const m = trimmed.match(/(\d+)\s*m/i);
   187‚Üí    const sec = trimmed.match(/(\d+)\s*s/i);
   188‚Üí    if (h || m || sec) {
   189‚Üí      let total = 0;
   190‚Üí      if (h) total += parseInt(h[1], 10) * 3600;
   191‚Üí      if (m) total += parseInt(m[1], 10) * 60;
   192‚Üí      if (sec) total += parseInt(sec[1], 10);
   193‚Üí      return total;
   194‚Üí    }
   195‚Üí  }
   196‚Üí  const parts = trimmed.split(":").map((p) => parseInt(p, 10));
   197‚Üí  if (parts.some((n) => isNaN(n))) return undefined;
   198‚Üí  // BARE NUMBER ‚Üí MINUTES (so "5" means 5 min, not 5 sec).
   199‚Üí  // This matches how a human naturally thinks about video timestamps:
   200‚Üí  // "skip to 5" = the 5-minute mark, not 5 seconds in.
   201‚Üí  if (parts.length === 1) return parts[0] * 60;
   202‚Üí  if (parts.length === 2) return parts[0] * 60 + parts[1];
   203‚Üí  if (parts.length === 3) return parts[0] * 3600 + parts[1] * 60 + parts[2];
   204‚Üí  return undefined;
   205‚Üí}

   207‚Üíexport function formatTime(s: number): string {
   208‚Üí  const h = Math.floor(s / 3600);
   209‚Üí  const m = Math.floor((s % 3600) / 60);
   210‚Üí  const sec = Math.floor(s % 60);
   211‚Üí  const pad = (n: number) => n.toString().padStart(2, "0");
   212‚Üí  return h > 0 ? `${h}:${pad(m)}:${pad(sec)}` : `${m}:${pad(sec)}`;
   213‚Üí}

   658‚Üífunction parseJson3Transcript(text: string): TranscriptSegment[] {
   659‚Üí  const json = JSON.parse(text);
   660‚Üí  if (!Array.isArray(json.events)) return [];
   661‚Üí  return json.events
   662‚Üí    .filter(
   663‚Üí      (e: any) => e.segs && Array.isArray(e.segs) && e.segs.length > 0
   664‚Üí    )
   665‚Üí    .map((e: any) => ({
   666‚Üí      start: (e.tStartMs ?? 0) / 1000,
   667‚Üí      dur: (e.dDurationMs ?? 0) / 1000,
   668‚Üí      text: e.segs
   669‚Üí        .map((s: any) => (typeof s.utf8 === "string" ? s.utf8 : ""))
   670‚Üí        .join("")
   671‚Üí        .replace(/\n/g, " ")
   672‚Üí        .trim(),
   673‚Üí    }))
   674‚Üí    .filter((s: TranscriptSegment) => s.text.length > 0);

   677‚Üífunction parseXmlTranscript(text: string): TranscriptSegment[] {
   678‚Üí  const segments: TranscriptSegment[] = [];
   679‚Üí  const srv3Regex = /<p\s+t="(\d+)"\s+d="(\d+)"[^>]*>([\s\S]*?)<\/p>/g;
   680‚Üí  let m: RegExpExecArray | null;
   681‚Üí  while ((m = srv3Regex.exec(text)) !== null) {
   682‚Üí    const start = parseInt(m[1], 10) / 1000;
   683‚Üí    const dur = parseInt(m[2], 10) / 1000;
   684‚Üí    const inner = m[3];
   685‚Üí    let segText = "";
   686‚Üí    const sRegex = /<s[^>]*>([^<]*)<\/s>/g;
   687‚Üí    let sMatch: RegExpExecArray | null;
   688‚Üí    while ((sMatch = sRegex.exec(inner)) !== null) {
   689‚Üí      segText += sMatch[1];
   690‚Üí    }
   691‚Üí    if (!segText) segText = inner.replace(/<[^>]+>/g, "");
   692‚Üí    segText = decodeEntities(segText).replace(/\n/g, " ").trim();
   693‚Üí    if (segText) segments.push({ start, dur, text: segText });
   694‚Üí  }
   695‚Üí  if (segments.length > 0) return segments;
   696‚Üí
   697‚Üí  const xmlRegex =
   698‚Üí    /<text\s+start="([\d.]+)"\s+dur="([\d.]+)"[^>]*>([\s\S]*?)<\/text>/g;
   699‚Üí  while ((m = xmlRegex.exec(text)) !== null) {
   700‚Üí    const start = parseFloat(m[1]);
   701‚Üí    const dur = parseFloat(m[2]);
   702‚Üí    const segText = decodeEntities(m[3])
   703‚Üí      .replace(/<[^>]+>/g, "")
   704‚Üí      .replace(/\n/g, " ")
   705‚Üí      .trim();
   706‚Üí    if (segText) segments.push({ start, dur, text: segText });
   707‚Üí  }
   708‚Üí  return segments;
   709‚Üí}

   754‚Üí *
   755‚Üí * Throws a `BotBlockedError` (with `code === "BOT_BLOCKED"`) if any strategy
   756‚Üí * reported a bot-block signature. Throws a generic Error otherwise.
   757‚Üí */
   758‚Üíexport async function fetchTranscriptWithRetry(
   759‚Üí  videoId: string

   763‚Üí    return cached.segments;
   764‚Üí  }
   765‚Üí
   766‚Üí  const trackStrategies: {
   767‚Üí    name: string;
   768‚Üí    fn: (id: string) => Promise<CaptionTrack[] | null>;
   769‚Üí  }[] = [
   770‚Üí    {
   771‚Üí      name: "ANDROID player API",
   772‚Üí      fn: fetchCaptionTracksViaAndroidPlayer,
   773‚Üí    },
   774‚Üí    {

   776‚Üí      fn: fetchCaptionTracksViaWatchPage,
   777‚Üí    },
   778‚Üí  ];
   779‚Üí
   780‚Üí  const directStrategies: {
   781‚Üí    name: string;
   782‚Üí    fn: (id: string) => Promise<TranscriptSegment[] | null>;
   783‚Üí  }[] = [
   784‚Üí    {
   785‚Üí      name: "youtube-transcript library",
   786‚Üí      fn: fetchTranscriptViaYoutubeTranscriptLib,
   787‚Üí    },
   788‚Üí    {
   789‚Üí      name: "youtubei.js (Innertube)",
   790‚Üí      fn: fetchTranscriptViaInnertube,
   791‚Üí    },
   792‚Üí  ];
   793‚Üí

   800‚Üí    );
   801‚Üí  }

   806‚Üí

   893‚Üí
   894‚Üí  transcriptCache.set(videoId, {
   895‚Üí    segments,

   898‚Üí
   899‚Üí  return segments;
   900‚Üí}

   902‚Üí/**
   903‚Üí * Parse a user-pasted transcript. Accepts:
   904‚Üí *   - "MM:SS text" or "HH:MM:SS text" (one per line)
   905‚Üí *   - "[MM:SS] text"
   906‚Üí *   - "1:23:45 text"
   907‚Üí *   - Plain text (no timestamps) ‚Äî each non-empty line becomes a segment
   908‚Üí *     with a sequential 5-second offset.
   909‚Üí */
   910‚Üíexport function parseUserTranscript(raw: string): {
   911‚Üí  segments: TranscriptSegment[];
   912‚Üí  hasTimestamps: boolean;
   913‚Üí} {
   914‚Üí  const lines = raw.split(/\r?\n/).map((l) => l.trim()).filter(Boolean);
   915‚Üí  if (lines.length === 0) return { segments: [], hasTimestamps: false };
   916‚Üí
   917‚Üí  const segments: TranscriptSegment[] = [];
   918‚Üí  let lastStart = 0;
   919‚Üí  let hasTimestamps = false;
   920‚Üí  const tsRegex = /^\[?(\d{1,2}:\d{2}(?::\d{2})?)\]?\s*(.*)$/;
   921‚Üí
   922‚Üí  for (const line of lines) {
   923‚Üí    const m = line.match(tsRegex);
   924‚Üí    if (m) {
   925‚Üí      const parts = m[1].split(":").map((p) => parseInt(p, 10));
   926‚Üí      let sec: number;
   927‚Üí      if (parts.length === 2) sec = parts[0] * 60 + parts[1];
   928‚Üí      else if (parts.length === 3) sec = parts[0] * 3600 + parts[1] * 60 + parts[2];
   929‚Üí      else sec = 0;
   930‚Üí      lastStart = sec;
   931‚Üí      hasTimestamps = true;
   932‚Üí      if (m[2]) {
   933‚Üí        segments.push({ start: sec, dur: 5, text: m[2].trim() });
   934‚Üí      }
   935‚Üí    } else {
   936‚Üí      segments.push({ start: lastStart, dur: 5, text: line });
   937‚Üí      lastStart += 5;
   938‚Üí    }
   939‚Üí  }
   940‚Üí
   941‚Üí  return { segments, hasTimestamps };
   942‚Üí}
