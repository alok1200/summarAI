    54→/**
    55→ * When the conversation is in "Ask about video" mode, the client sends this
    56→ * object so we can inject the video transcript as system context. The strict
    57→ * prompt below tells the model to ONLY answer questions that are answerable
    58→ * from the transcript — anything off-topic gets a fixed "not in this video" reply.
    59→ *
    60→ * Two modes:
    61→ *   - Short video: `transcript` is set (string). Whole transcript injected.
    62→ *   - Long video: `chunks` + `topicIndex` are set. We do RETRIEVAL — find
    63→ *     the most relevant chunks for the user's latest question and inject
    64→ *     only those (so even a 50-hour video can be queried within context).
    65→ */
    66→interface VideoContextPayload {
    67→  url: string;
    68→  videoId: string;
    69→  title: string;
    70→  author: string;
    71→  /** Short-video mode: full transcript text. */
    72→  transcript?: string;
    73→  /** Long-video mode: chunks array. */
    74→  chunks?: VideoChunkPayload[];
    75→  /** Long-video mode: topic index for retrieval. */
    76→  topicIndex?: string;
    77→  startTime?: number;
    78→  endTime?: number;
    79→  /**
    80→   * Optional: language for AI responses during ask-about-video Q&A.
    81→   * Empty/undefined = default (English). Persisted across the conversation
    82→   * so every follow-up question is answered in the chosen language.
    83→   */
    84→  language?: string;
    85→}

   103→      : transcript;

   105→    `You are an expert AI tutor that answers the user's questions STRICTLY based ` +
   106→    `on the transcript of a YouTube video they have loaded. Think of yourself as ` +
   107→    `the world's leading expert on THIS specific video — your only job is to help ` +
   108→    `the user understand its content IN FULL DETAIL.\n\n` +
   109→    `VIDEO METADATA:\n` +
   110→    `- Title: ${ctx.title}\n` +
   111→    `- Channel: ${ctx.author}\n` +
   112→    `- URL: ${ctx.url}\n` +
   113→    (ctx.startTime !== undefined && ctx.endTime !== undefined
   114→      ? `- Loaded time range: ${ctx.startTime}s – ${ctx.endTime}s\n`
   115→      : "") +
   116→    `\nVIDEO TRANSCRIPT (with timestamps):\n"""\n${truncated}\n"""\n\n` +
   117→    `STRICT RULES:\n` +
   118→    `1. Answer ONLY questions that can be answered from the transcript above.\n` +
   119→    `2. If the user asks about ANYTHING not covered in the transcript — including ` +
   120→    `general knowledge, current events, other videos, coding contests, homework, ` +

   122→    `message (no other text):\n\n` +
   123→    `   "${VIDEO_OFF_TOPIC_REPLY}"\n\n` +
   124→    `3. Do NOT use your general knowledge to fill in gaps. If the transcript ` +
   125→    `doesn't say it, you don't say it.\n` +
   126→    `4. When answering, you may quote or paraphrase the transcript. ALWAYS reference ` +
   127→    `timestamps in square brackets so the user can locate the source moment — ` +
   128→    `copy them EXACTLY as they appear in the transcript (e.g. [3:25] or [1:25:30]).\n\n` +
   129→    TIMESTAMP_RULES + `\n\n` +
   130→    `ANSWER STYLE — BE EXHAUSTIVE & DETAILED:\n` +
   131→    `- Give a COMPLETE answer that covers every aspect of the question mentioned ` +
   132→    `  anywhere in the transcript. Do not give a one-line answer when the video ` +
   133→    `  discusses the topic in depth.\n` +

   153→/**
   154→ * RETRIEVAL step for LONG videos.
   155→ *
   156→ * TWO-TIER STRATEGY:
   157→ *   1. VECTOR SEARCH (primary): embed the user's question, do cosine
   158→ *      similarity search against chunk embeddings in the DB. This is
   159→ *      ~1000× cheaper and ~100× faster than the LLM-as-retriever fallback.
   160→ *      Requires the transcript to be persisted + embedded (see
   161→ *      /api/youtube-load → persistTranscript → embedTranscriptInBackground).
   162→ *
   163→ *   2. LLM-AS-RETRIEVER (fallback): if vector search isn't available
   164→ *      (transcript not in DB, embeddings not generated yet, or embedding
   165→ *      the question failed), ask the LLM to pick chunk numbers from the
   166→ *      topic index. This is the original retrieval mechanism and works
   167→ *      without any vector infrastructure — slower + more expensive but
   168→ *      always available as long as the topic index exists.
   169→ *
   170→ * The fallback guarantees the chat works even on first load (before
   171→ * background embedding completes) and if the embedding model fails to
   172→ * download (e.g., offline environment).
   173→ *
   174→ * Returns:
   175→ *   - chunks: the selected chunks to inject into the system prompt
   176→ *   - offTopic: true if NO chunks were selected (question is unrelated
   177→ *     to the video) — caller should reply with the off-topic message
   178→ *     instead of asking the LLM
   179→ *   - source: "vector" | "llm" | "fallback" — for logging/metrics
   180→ */
   181→async function retrieveRelevantChunks(
   182→  userQuestion: string,
   183→  ctx: VideoContextPayload,
   184→  userId: string
   185→): Promise<{ chunks: VideoChunkPayload[]; offTopic: boolean; source: "vector" | "llm" | "fallback" }> {
   186→  const allChunks = ctx.chunks ?? [];
   187→  if (allChunks.length === 0) {
   188→    return { chunks: [], offTopic: true, source: "fallback" };
   189→  }
   190→
   191→  // ---------- TIER 1: Vector search ----------
   192→  // Try vector search first. This requires:
   193→  //   1. The video has a Transcript row in the DB for this user
   194→  //   2. At least some chunks have embeddings (transcript.embedded = true)
   195→  //   3. embedText() succeeds for the user's question
   196→  //
   197→  // If any of these fail, fall through to LLM-as-retriever below.
   198→  try {
   199→    const transcript = await db.transcript.findUnique({
   200→      where: { userId_videoId: { userId, videoId: ctx.videoId } },
   201→      select: { id: true, embedded: true, chunkCount: true },
   202→    });
   203→
   204→    if (transcript && transcript.embedded) {
   205→      const queryEmbedding = await embedText(userQuestion);
   206→      if (queryEmbedding) {
   207→        const results: VectorSearchResult[] = await vectorRetrieve(
   208→          transcript.id,
   209→          queryEmbedding,
   210→          3
   211→        );
   212→
   213→        if (results.length > 0) {
   214→          // Map vector-search results back to VideoChunkPayload[] using
   215→          // chunkIndex. The chunks payload from the client is 1-indexed
   216→          // (c.index field), while our DB chunks are 0-indexed (chunkIndex
   217→          // column). Convert: DB chunkIndex 0 → client chunk 1.
   218→          const selected = results
   219→            .map((r) => allChunks.find((c) => c.index === r.chunkIndex + 1))
   220→            .filter(Boolean) as VideoChunkPayload[];
   221→
   222→          if (selected.length > 0) {
   223→            logger.info("chat.retrieval.vector_hit", {
   224→              userId,
   225→              videoId: ctx.videoId,
   226→              topScore: results[0].score,
   227→              selectedCount: selected.length,
   228→            });
   229→            return { chunks: selected, offTopic: false, source: "vector" };
   230→          }
   231→        }
   232→
   233→        // Vector search returned 0 results — embeddings exist but nothing
   234→        // matched. This is rare (would mean every chunk scored below the
   235→        // implicit filter). Treat as off-topic ONLY if the LLM agrees.
   236→        // Fall through to LLM-as-retriever for a second opinion.
   237→        logger.info("chat.retrieval.vector_empty", {
   238→          userId,
   239→          videoId: ctx.videoId,
   240→        });
   241→      } else {
   242→        logger.warn("chat.retrieval.embed_failed", {
   243→          userId,
   244→          videoId: ctx.videoId,
   245→        });
   246→      }
   247→    }
   248→  } catch (err) {
   249→    // DB error or unexpected — log and fall through to LLM-as-retriever.
   250→    logger.error("chat.retrieval.vector_error", {
   251→      userId,
   252→      videoId: ctx.videoId,
   253→      error: err instanceof Error ? err.message : String(err),
   254→    });
   255→  }
   256→
   257→  // ---------- TIER 2: LLM-as-retriever (fallback) ----------
   258→  return retrieveRelevantChunksLLM(userQuestion, ctx);
   259→}
   260→
   261→/**
   262→ * LLM-as-retriever fallback — the original retrieval mechanism.
   263→ *
   264→ * Given the user's latest question and the video's topic index, ask the
   265→ * LLM to pick the most relevant chunk numbers. Returns an array of chunk
   266→ * indexes (1-indexed) to inject into context.
   267→ *
   268→ * If the LLM decides the question is off-topic for the whole video, it
   269→ * returns an empty array — and we reply with the off-topic message without
   270→ * even loading any chunks.
   271→ */
   272→async function retrieveRelevantChunksLLM(
   273→  userQuestion: string,
   274→  ctx: VideoContextPayload
   275→): Promise<{ chunks: VideoChunkPayload[]; offTopic: boolean; source: "llm" | "fallback" }> {
   276→  const allChunks = ctx.chunks ?? [];
   277→
   278→  const systemPrompt =
   279→    `You are a retrieval system for a long YouTube video. The user has loaded ` +
   280→    `a video with ${allChunks.length} chunks. Each chunk covers ~5-10 minutes ` +
   281→    `of the video. Below is the TOPIC INDEX — a list of topics covered in each ` +
   282→    `chunk, with timestamps.\n\n` +
   283→    `Your job: given the user's question, decide which chunks are most likely ` +
   284→    `to contain the answer. Return AT MOST 3 chunk numbers (the most relevant ` +
   285→    `ones), as a JSON array of integers.\n\n` +
   286→    `If NONE of the chunks seem relevant to the question (i.e., the question is ` +
   287→    `about something not covered anywhere in the video), return an empty array: ` +
   288→    `[]\n\n` +
   289→    `OUTPUT FORMAT: respond with ONLY a JSON array of integers, no other text. ` +
   290→    `Examples:\n` +
   291→    `  - [3, 7]      (chunks 3 and 7 are most relevant)\n` +
   292→    `  - [1]         (only chunk 1 is relevant)\n` +
   293→    `  - []          (question is off-topic for this video)\n\n` +
   294→    `TOPIC INDEX:\n${ctx.topicIndex ?? "(no topic index available)"}`;
   295→
   296→  const userMessage =
   297→    `User question: "${userQuestion}"\n\n` +
   298→    `Return the JSON array of relevant chunk numbers now.`;
   299→
   300→  let responseText: string;
   301→  try {
   302→    responseText = await chatComplete([
   303→      { role: "system", content: systemPrompt },
   304→      { role: "user", content: userMessage },
   305→    ]);
   306→  } catch (err) {
   307→    console.error("[chat] retrieval LLM call failed:", err);
   308→    // Fallback: just use the first 2 chunks
   309→    return { chunks: allChunks.slice(0, 2), offTopic: false, source: "fallback" };
   310→  }
   311→
   312→  // Parse the JSON array from the response
   313→  const match = responseText.match(/\[[\s\S]*\]/);
   314→  if (!match) {
   315→    console.warn("[chat] retrieval response had no JSON array:", responseText.slice(0, 200));
   316→    return { chunks: allChunks.slice(0, 2), offTopic: false, source: "fallback" };
   317→  }
   318→  let indexes: number[];
   319→  try {
   320→    indexes = JSON.parse(match[0]);
   321→  } catch {
   322→    return { chunks: allChunks.slice(0, 2), offTopic: false, source: "fallback" };
   323→  }
   324→  if (!Array.isArray(indexes) || indexes.length === 0) {
   325→    return { chunks: [], offTopic: true, source: "llm" };
   326→  }
   327→
   328→  // De-dup, clamp to valid range, take top 3
   329→  const unique = Array.from(new Set(indexes))
   330→    .filter((n) => typeof n === "number" && n >= 1 && n <= allChunks.length)
   331→    .slice(0, 3);
   332→  if (unique.length === 0) {
   333→    return { chunks: [], offTopic: true, source: "llm" };
   334→  }
   335→
   336→  const selected = unique.map(
   337→    (n) => allChunks.find((c) => c.index === n) ?? allChunks[n - 1]
   338→  );
   339→  return { chunks: selected.filter(Boolean), offTopic: false, source: "llm" };
   340→}

   400→function buildMessageText(m: ChatMessagePayload): string {
   401→  if (!m.attachments || m.attachments.length === 0) return m.content;
   402→  const textParts: string[] = [];
   403→  if (m.content.trim()) textParts.push(m.content);
   404→  for (const a of m.attachments) {
   405→    if (a.kind === "text" && a.textContent) {
   406→      textParts.push(
   407→        `\n\n--- Attached file: ${a.name} ---\n${a.textContent}\n--- end of ${a.name} ---`
   408→      );
   409→    } else if (a.kind === "image") {
   410→      textParts.push(`[Image attached: ${a.name}]`);
   411→    } else {
   412→      textParts.push(`[File attached: ${a.name}]`);
   413→    }
   414→  }
   415→  return textParts.join("");
   416→}

   418→export async function POST(req: NextRequest) {
   419→  // ---------- AUTH ----------
   420→  const guard = await requireAuth(req);
   421→  if (!guard.ok) return guard.response;
   422→  const { user } = guard;
   423→
   424→  // ---------- RATE LIMIT (per-user, 10/min by default) ----------
   425→  const rl = rateLimit(aiRateLimitConfig(user.id, "chat"));
   426→  if (!rl.ok) {
   427→    logger.warn("chat.rate_limited", { userId: user.id });
   428→    return rl.response;
   429→  }
   430→
   431→  // ---------- BODY (size-capped, JSON-validated) ----------
   432→  const bodyResult = await readJsonBody<{
   433→    messages?: ChatMessagePayload[];
   434→    videoContext?: VideoContextPayload;
   435→    systemPrompt?: string;
   436→  }>(req);
   437→  if (!bodyResult.ok) return bodyResult.response;
   438→  const body = bodyResult.value;
   439→
   440→  const requestId = req.headers.get("x-request-id") ?? "no-req-id";
   441→  logger.info("chat.request", {
   442→    userId: user.id,
   443→    requestId,
   444→    messageCount: body.messages?.length ?? 0,
   445→    hasVideoContext: !!body.videoContext,
   446→    hasImages:
   447→      body.messages?.some((m) =>
   448→        m.attachments?.some((a) => a.kind === "image" && a.dataUrl)
   449→      ) ?? false,
   450→  });
   451→
   452→  try {
   453→    const messages: ChatMessagePayload[] = body.messages ?? [];
   454→    const videoContext: VideoContextPayload | undefined = body.videoContext;
   455→
   456→    // Determine video mode: long-video mode requires chunks + topicIndex.
   457→    const isLongVideoMode =
   458→      videoContext &&
   459→      !videoContext.transcript &&
   460→      videoContext.chunks &&
   461→      videoContext.chunks.length > 0 &&
   462→      !!videoContext.topicIndex;
   463→
   464→    let systemPrompt: string;
   465→    let offTopicReply: string | null = null;
   466→
   467→    if (videoContext && isLongVideoMode) {
   468→      // ---------- LONG VIDEO: retrieval-augmented Q&A ----------
   469→      // Find the latest user question (text only, no images).
   470→      const lastUserText = [...messages]
   471→        .reverse()
   472→        .find((m) => m.role === "user" && m.content.trim());
   473→      const userQuestion = lastUserText?.content ?? "";
   474→
   475→      const retrieved = await retrieveRelevantChunks(userQuestion, videoContext, user.id);
   476→      if (retrieved.offTopic || retrieved.chunks.length === 0) {
   477→        // Skip the LLM call entirely and short-circuit with the off-topic reply.
   478→        offTopicReply = VIDEO_OFF_TOPIC_REPLY;
   479→        systemPrompt = "";
   480→      } else {
   481→        systemPrompt = buildLongVideoSystemPrompt(videoContext, retrieved.chunks);
   482→      }
   483→    } else if (videoContext && videoContext.transcript) {
   484→      // ---------- SHORT VIDEO: inject whole transcript ----------
   485→      systemPrompt = buildShortVideoSystemPrompt(videoContext);
   486→    } else {
   487→      systemPrompt =
   488→        body.systemPrompt ??
   489→        ("You are a world-class AI assistant — friendly, precise, and relentlessly helpful. " +
   490→        "Your single goal is to FULLY SOLVE the user's problem and leave them satisfied — never hand them a one-liner when they need a real answer.\n\n" +
   491→        "CORE PRINCIPLES:\n" +
   492→        "- SOLVE the problem. When the user asks how to do something, give them a working, complete solution they can act on immediately — not a hint and not a partial answer.\n" +
