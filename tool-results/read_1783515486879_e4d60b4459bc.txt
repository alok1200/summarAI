   153→   153→/**
   154→   154→ * RETRIEVAL step for LONG videos.
   155→   155→ *
   156→   156→ * TWO-TIER STRATEGY:
   157→   157→ *   1. VECTOR SEARCH (primary): embed the user's question, do cosine
   158→   158→ *      similarity search against chunk embeddings in the DB. This is
   159→   159→ *      ~1000× cheaper and ~100× faster than the LLM-as-retriever fallback.
   160→   160→ *      Requires the transcript to be persisted + embedded (see
   161→   161→ *      /api/youtube-load → persistTranscript → embedTranscriptInBackground).
   162→   162→ *
   163→   163→ *   2. LLM-AS-RETRIEVER (fallback): if vector search isn't available
   164→   164→ *      (transcript not in DB, embeddings not generated yet, or embedding
   165→   165→ *      the question failed), ask the LLM to pick chunk numbers from the
   166→   166→ *      topic index. This is the original retrieval mechanism and works
   167→   167→ *      without any vector infrastructure — slower + more expensive but
   168→   168→ *      always available as long as the topic index exists.
   169→   169→ *
   170→   170→ * The fallback guarantees the chat works even on first load (before
   171→   171→ * background embedding completes) and if the embedding model fails to
   172→   172→ * download (e.g., offline environment).
   173→   173→ *
   174→   174→ * Returns:
   175→   175→ *   - chunks: the selected chunks to inject into the system prompt
   176→   176→ *   - offTopic: true if NO chunks were selected (question is unrelated
   177→   177→ *     to the video) — caller should reply with the off-topic message
   178→   178→ *     instead of asking the LLM
   179→   179→ *   - source: "vector" | "llm" | "fallback" — for logging/metrics
   180→   180→ */
   181→   181→async function retrieveRelevantChunks(
   182→   182→  userQuestion: string,
   183→   183→  ctx: VideoContextPayload,
   184→   184→  userId: string
   185→   185→): Promise<{ chunks: VideoChunkPayload[]; offTopic: boolean; source: "vector" | "llm" | "fallback" }> {
   186→   186→  const allChunks = ctx.chunks ?? [];
   187→   187→  if (allChunks.length === 0) {
   188→   188→    return { chunks: [], offTopic: true, source: "fallback" };
   189→   189→  }
   190→   190→
   191→   191→  // ---------- TIER 1: Vector search ----------
   192→   192→  // Try vector search first. This requires:
   193→   193→  //   1. The video has a Transcript row in the DB for this user
   194→   194→  //   2. At least some chunks have embeddings (transcript.embedded = true)
   195→   195→  //   3. embedText() succeeds for the user's question
   196→   196→  //
   197→   197→  // If any of these fail, fall through to LLM-as-retriever below.
   198→   198→  try {
   199→   199→    const transcript = await db.transcript.findUnique({
   200→   200→      where: { userId_videoId: { userId, videoId: ctx.videoId } },
   201→   201→      select: { id: true, embedded: true, chunkCount: true },
   202→   202→    });
   203→   203→
   204→   204→    if (transcript && transcript.embedded) {
   205→   205→      const queryEmbedding = await embedText(userQuestion);
   206→   206→      if (queryEmbedding) {
   207→   207→        const results: VectorSearchResult[] = await vectorRetrieve(
   208→   208→          transcript.id,
   209→   209→          queryEmbedding,
   210→   210→          3
   211→   211→        );
   212→   212→
   213→   213→        if (results.length > 0) {
   214→   214→          // Map vector-search results back to VideoChunkPayload[] using
   215→   215→          // chunkIndex. The chunks payload from the client is 1-indexed
   216→   216→          // (c.index field), while our DB chunks are 0-indexed (chunkIndex
   217→   217→          // column). Convert: DB chunkIndex 0 → client chunk 1.
   218→   218→          const selected = results
   219→   219→            .map((r) => allChunks.find((c) => c.index === r.chunkIndex + 1))
   220→   220→            .filter(Boolean) as VideoChunkPayload[];
   221→   221→
   222→   222→          if (selected.length > 0) {
   223→   223→            logger.info("chat.retrieval.vector_hit", {
   224→   224→              userId,
   225→   225→              videoId: ctx.videoId,
   226→   226→              topScore: results[0].score,
   227→   227→              selectedCount: selected.length,
   228→   228→            });
   229→   229→            return { chunks: selected, offTopic: false, source: "vector" };
   230→   230→          }
   231→   231→        }
   232→   232→
   233→   233→        // Vector search returned 0 results — embeddings exist but nothing
   234→   234→        // matched. This is rare (would mean every chunk scored below the
   235→   235→        // implicit filter). Treat as off-topic ONLY if the LLM agrees.
   236→   236→        // Fall through to LLM-as-retriever for a second opinion.
   237→   237→        logger.info("chat.retrieval.vector_empty", {
   238→   238→          userId,
   239→   239→          videoId: ctx.videoId,
   240→   240→        });
   241→   241→      } else {
   242→   242→        logger.warn("chat.retrieval.embed_failed", {
   243→   243→          userId,
   244→   244→          videoId: ctx.videoId,
   245→   245→        });
   246→   246→      }
   247→   247→    }
   248→   248→  } catch (err) {
   249→   249→    // DB error or unexpected — log and fall through to LLM-as-retriever.
   250→   250→    logger.error("chat.retrieval.vector_error", {
   251→   251→      userId,
   252→   252→      videoId: ctx.videoId,
   253→   253→      error: err instanceof Error ? err.message : String(err),
   254→   254→    });
   255→   255→  }
   256→   256→
   257→   257→  // ---------- TIER 2: LLM-as-retriever (fallback) ----------
   258→   258→  return retrieveRelevantChunksLLM(userQuestion, ctx);
   259→   259→}
   260→   260→
   261→   261→/**
   262→   262→ * LLM-as-retriever fallback — the original retrieval mechanism.
   263→   263→ *
   264→   264→ * Given the user's latest question and the video's topic index, ask the
   265→   265→ * LLM to pick the most relevant chunk numbers. Returns an array of chunk
   266→   266→ * indexes (1-indexed) to inject into context.
   267→   267→ *
   268→   268→ * If the LLM decides the question is off-topic for the whole video, it
   269→   269→ * returns an empty array — and we reply with the off-topic message without
   270→   270→ * even loading any chunks.
   271→   271→ */
   272→   272→async function retrieveRelevantChunksLLM(
   273→   273→  userQuestion: string,
   274→   274→  ctx: VideoContextPayload
   275→   275→): Promise<{ chunks: VideoChunkPayload[]; offTopic: boolean; source: "llm" | "fallback" }> {
   276→   276→  const allChunks = ctx.chunks ?? [];
   277→   277→
   278→   278→  const systemPrompt =
   279→   279→    `You are a retrieval system for a long YouTube video. The user has loaded ` +
   280→   280→    `a video with ${allChunks.length} chunks. Each chunk covers ~5-10 minutes ` +
   281→   281→    `of the video. Below is the TOPIC INDEX — a list of topics covered in each ` +
   282→   282→    `chunk, with timestamps.\n\n` +
   283→   283→    `Your job: given the user's question, decide which chunks are most likely ` +
   284→   284→    `to contain the answer. Return AT MOST 3 chunk numbers (the most relevant ` +
   285→   285→    `ones), as a JSON array of integers.\n\n` +
   286→   286→    `If NONE of the chunks seem relevant to the question (i.e., the question is ` +
   287→   287→    `about something not covered anywhere in the video), return an empty array: ` +
   288→   288→    `[]\n\n` +
   289→   289→    `OUTPUT FORMAT: respond with ONLY a JSON array of integers, no other text. ` +
   290→   290→    `Examples:\n` +
   291→   291→    `  - [3, 7]      (chunks 3 and 7 are most relevant)\n` +
   292→   292→    `  - [1]         (only chunk 1 is relevant)\n` +
   293→   293→    `  - []          (question is off-topic for this video)\n\n` +
   294→   294→    `TOPIC INDEX:\n${ctx.topicIndex ?? "(no topic index available)"}`;
   295→   295→
   296→   296→  const userMessage =
   297→   297→    `User question: "${userQuestion}"\n\n` +
   298→   298→    `Return the JSON array of relevant chunk numbers now.`;
   299→   299→
   300→   300→  let responseText: string;
   301→   301→  try {
   302→   302→    responseText = await chatComplete([
   303→   303→      { role: "system", content: systemPrompt },
   304→   304→      { role: "user", content: userMessage },
   305→   305→    ]);
   306→   306→  } catch (err) {
   307→   307→    console.error("[chat] retrieval LLM call failed:", err);
   308→   308→    // Fallback: just use the first 2 chunks
   309→   309→    return { chunks: allChunks.slice(0, 2), offTopic: false, source: "fallback" };
   310→   310→  }
   311→   311→
   312→   312→  // Parse the JSON array from the response
   313→   313→  const match = responseText.match(/\[[\s\S]*\]/);
   314→   314→  if (!match) {
   315→   315→    console.warn("[chat] retrieval response had no JSON array:", responseText.slice(0, 200));
   316→   316→    return { chunks: allChunks.slice(0, 2), offTopic: false, source: "fallback" };
   317→   317→  }
   318→   318→  let indexes: number[];
   319→   319→  try {
   320→   320→    indexes = JSON.parse(match[0]);
   321→   321→  } catch {
   322→   322→    return { chunks: allChunks.slice(0, 2), offTopic: false, source: "fallback" };
   323→   323→  }
   324→   324→  if (!Array.isArray(indexes) || indexes.length === 0) {
   325→   325→    return { chunks: [], offTopic: true, source: "llm" };
   326→   326→  }
   327→   327→
   328→   328→  // De-dup, clamp to valid range, take top 3
   329→   329→  const unique = Array.from(new Set(indexes))
   330→   330→    .filter((n) => typeof n === "number" && n >= 1 && n <= allChunks.length)
   331→   331→    .slice(0, 3);
   332→   332→  if (unique.length === 0) {
   333→   333→    return { chunks: [], offTopic: true, source: "llm" };
   334→   334→  }
   335→   335→
   336→   336→  const selected = unique.map(
   337→   337→    (n) => allChunks.find((c) => c.index === n) ?? allChunks[n - 1]
   338→   338→  );
   339→   339→  return { chunks: selected.filter(Boolean), offTopic: false, source: "llm" };
   340→   340→}

   400→   400→function buildMessageText(m: ChatMessagePayload): string {
   401→   401→  if (!m.attachments || m.attachments.length === 0) return m.content;
   402→   402→  const textParts: string[] = [];
   403→   403→  if (m.content.trim()) textParts.push(m.content);
   404→   404→  for (const a of m.attachments) {
   405→   405→    if (a.kind === "text" && a.textContent) {
   406→   406→      textParts.push(
   407→   407→        `\n\n--- Attached file: ${a.name} ---\n${a.textContent}\n--- end of ${a.name} ---`
   408→   408→      );
   409→   409→    } else if (a.kind === "image") {
   410→   410→      textParts.push(`[Image attached: ${a.name}]`);
   411→   411→    } else {
   412→   412→      textParts.push(`[File attached: ${a.name}]`);
   413→   413→    }
   414→   414→  }
   415→   415→  return textParts.join("");
   416→   416→}
